1. 编写程序，输入一个一元二次方程的系数a,b,c，输出该方程的2个根
```c++
#include <cmath>
#include <cstdio>
#include <iostream>

int main()
{
    double a, b, c;
    std::cin >> a >> b >> c;
    double delta = b * b - 4 * a * c;
    if (delta < 0)
    {
        std::cout << "No real root" << std::endl;
    }
    else if (delta == 0)
    {
        std::cout << "x1=x2=" << -b / (2 * a) << std::endl;
    }
    else
    {
        std::cout << "x1=" << (-b + sqrt(delta)) / (2 * a) << std::endl;
        std::cout << "x2=" << (-b - sqrt(delta)) / (2 * a) << std::endl;
    }

    return 0;
}
```

3. 输入一个正整数判断其是否是回文数，如果是则输出YES，否则输出NO。
```c++
#include <iostream>
using namespace std;
int main()
{
    int n;
    cin >> n;
    int m = n;
    int sum = 0;
    while (n != 0)
    {
        sum = sum * 10 + n % 10;
        n = n / 10;
    }
    if (sum == m)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;
    return 0;
}
```
